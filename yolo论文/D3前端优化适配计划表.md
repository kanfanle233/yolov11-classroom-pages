#### 任务 2.2：D3 雷达图（Verifier 证据可视化）

**目标**：四维证据 [visual_score, text_score, uq(逆), w_visual, w_audio] 的雷达图

**当前状态**：❌ 未实现

**改造方案**（纯D3实现，无额外依赖）：

```javascript
function drawRadarChart(evidence) {
  if (!evidence) return;
  const container = document.getElementById('radar-chart');
  container.innerHTML = '';
  
  const axes = [
    {name: '视觉分数', value: evidence.visual_score || 0, max: 1},
    {name: '文本分数', value: evidence.text_score || 0, max: 1},
    {name: '跟踪稳定性', value: 1 - (evidence.uq_score || 0), max: 1},
    {name: '视觉权重', value: evidence.w_visual || 0, max: 1},
    {name: '音频权重', value: evidence.w_audio || 0, max: 1},
  ];
  
  const w = 200, h = 200, r = 80;
  const cx = w/2, cy = h/2;
  const angleSlice = (2 * Math.PI) / axes.length;
  
  const svg = d3.select(container).append('svg')
    .attr('viewBox', `0 0 ${w} ${h}`).style('width','200px').style('height','200px');
  
  const g = svg.append('g').attr('transform', `translate(${cx},${cy})`);
  
  // 背景网格
  [0.25, 0.5, 0.75, 1.0].forEach(level => {
    const pts = axes.map((_, i) => {
      const a = i * angleSlice - Math.PI/2;
      return [r * level * Math.cos(a), r * level * Math.sin(a)];
    });
    g.append('polygon').attr('points', pts.map(p=>p.join(',')).join(' '))
      .attr('fill','none').attr('stroke','#ddd').attr('stroke-width',0.5);
  });
  
  // 轴线
  axes.forEach((_, i) => {
    const a = i * angleSlice - Math.PI/2;
    g.append('line').attr('x1',0).attr('y1',0)
      .attr('x2', r*Math.cos(a)).attr('y2', r*Math.sin(a))
      .attr('stroke','#ccc').attr('stroke-width',0.5);
  });
  
  // 数据多边形
  const dataPts = axes.map((d, i) => {
    const a = i * angleSlice - Math.PI/2;
    const val = Math.max(0, Math.min(1, d.value));
    return [r * val * Math.cos(a), r * val * Math.sin(a)];
  });
  g.append('polygon').attr('points', dataPts.map(p=>p.join(',')).join(' '))
    .attr('fill','rgba(9,132,227,0.3)').attr('stroke','#0984e3').attr('stroke-width',1.5);
  
  // 标签
  axes.forEach((d, i) => {
    const a = i * angleSlice - Math.PI/2;
    g.append('text')
      .attr('x', (r+18)*Math.cos(a)).attr('y', (r+18)*Math.sin(a))
      .attr('text-anchor','middle').attr('dominant-baseline','middle')
      .attr('font-size',9).attr('fill','#596b7d')
      .text(d.name);
  });
}
```

**工作量**：0.5天（纯JS函数 + 集成到证据面板）
